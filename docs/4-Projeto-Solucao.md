#### 4.3.3 Modelo Físico

CÓDIGO:
create table public.cliente (
  id bigint generated by default as identity primary key,
  nome text,
  telefone bigint,
  email text,
  "endereço" text,
  senha bigint,
  admin text
);

create table public.estoque (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default now()
);

create table public."loginRequet" (
  id bigint generated by default as identity primary key,
  email text,
  senha bigint
);

create table public."peças" (
  id bigint generated by default as identity primary key,
  nome text,
  codigo bigint,
  quantidade bigint,
  "preçoUnitario" bigint
);

create table public."serviço" (
  id bigint generated by default as identity primary key,
  cliente text,
  "tipoServiço" text,
  moto text,
  placa text,
  telefone text,
  data date,
  horario text,
  "observações" text
);

create table public."solicitaçãoServiço" (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default now(),
  status text
);

create table public."ordemServiço" (
  id bigint generated by default as identity primary key,
  "serviçoId" bigint references public."serviço"(id),
  "dataEmissão" date,
  status text,
  "descrição" text,
  "clienteId" bigint references public.cliente(id),
  "valorTotal" bigint,
  cliente text
);

create table public."peçasUsadas" (
  id bigint generated by default as identity primary key,
  "peçaId" bigint references public."peças"(id),
  "ordemServiçoId" bigint references public."ordemServiço"(id),
  quantidade bigint
);
